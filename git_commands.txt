# Git Commands for HR Detection GUI Project
# Run these commands in Anaconda Prompt

# 1. Navigate to project (Z: is already mapped)
cd "Z:\"

# 2. Initialize git repository (if not already done)
git init

# 3. Check status
git status

# 4. Add all files
git add .

# 5. Make first commit
git commit -m "Initial commit: Heart Rate Detection GUI"

# 6. Add GitHub remote (replace YOUR_USERNAME and REPO_NAME)
git remote add origin https://github.com/YOUR_USERNAME/REPO_NAME.git

# 7. Set main branch
git branch -M main

# 8. Push to GitHub
git push -u origin main
